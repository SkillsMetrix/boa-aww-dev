
import boto3
import json

s3= boto3.client('s3')

def lambda_handler(event,context):
    bucket= event['Records'][0]['s3']['bucket']['name']
    key= event['Records'][0]['s3']['object']['key']
    print(f"File Uploaded: {key} in bucket: {bucket}")

    return {
        'statusCode': 200,
        'body': json.dumps(f'Hello from Lambda!')
    }
